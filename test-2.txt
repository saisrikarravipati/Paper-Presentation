WITH first50 AS (
  SELECT created_date
  FROM your_table
  ORDER BY created_date
  LIMIT 50
),
bounds AS (
  SELECT
    MIN(created_date)        AS first_ts,
    MAX(created_date)        AS last_ts,
    COUNT(*)                 AS cnt
  FROM first50
)
SELECT
  -- average interval in seconds between first and last over the (cntâˆ’1) intervals
  EXTRACT(EPOCH FROM (last_ts - first_ts)) / (cnt - 1) AS avg_processing_seconds,
  (last_ts - first_ts) / (cnt - 1)                      AS avg_processing_interval
FROM bounds;
